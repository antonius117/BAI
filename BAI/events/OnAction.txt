# Nation that owns your debt goes bankrupt
country_event = {
	id = 70002
	title = "EVTNAME70002"
	desc = "EVTDESC70002"
	picture = "Bankruptcy"
	trigger = {
		NOT = { tag = FROM }
	}

	is_triggered_only = yes
	option = {
		name = "EVTOPTA70002"
		relation = { who = FROM value = -50 }
		random_country = {
			limit = {
				tag = FROM
				is_vassal = no
				THIS = { is_greater_power = yes }
			}
			add_casus_belli = {
				target = THIS
				type = gunboat
				months = 24
			}
		}
		random_country = {
			limit = {
				tag = FROM
				is_vassal = no
				part_of_sphere = no
				THIS = { is_greater_power = yes }
			}
			add_casus_belli = {
				target = THIS
				type = add_to_sphere
				months = 24
			}
		}
		random_country = {
			limit = {
				tag = FROM
				OR = {
					neighbour = THIS
					THIS = { ai = no }
				}
				is_disarmed = no
				is_vassal = no
				THIS = { is_greater_power = no }
			}
			add_casus_belli = {
				target = THIS
				type = cut_down_to_size
				months = 12
			}
		}
		ai_chance = {
			factor = 1
		}
	}
	
	option = {
		name = "Look the other way"
		relation = { who = FROM value = 50 }
		prestige = -5
		any_pop = {
			limit = {
				strata = rich
			}
			militancy = 2
		}
		ai_chance = {
			factor = 1
		}
	}
}
